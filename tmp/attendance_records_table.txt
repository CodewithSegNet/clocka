      {/* Attendance Records */}
      {filteredRecords.length === 0 ? (
        <div className="bg-white rounded-2xl p-16 text-center border border-gray-200 shadow-sm">
          <ClipboardList className="w-16 h-16 text-gray-300 mx-auto mb-4" />
          <p className="text-gray-500 text-lg mb-2">
            {groupedRecords.length === 0 ? 'No attendance records found' : 'No records match the current filters'}
          </p>
          <p className="text-sm text-gray-400">
            {groupedRecords.length === 0 
              ? 'Attendance records will appear here when parents clock in/out'
              : 'Try adjusting your filters to see more results'
            }
          </p>
          {(statusFilter !== 'all' || tagFilter !== 'all' || searchQuery) && (
            <button
              onClick={() => {
                setStatusFilter('all');
                setTagFilter('all');
                setSearchQuery('');
              }}
              className="mt-4 px-4 py-2 bg-indigo-600 text-white rounded-lg hover:bg-indigo-700 transition-colors text-sm font-medium"
            >
              Clear all filters
            </button>
          )}
        </div>
      ) : (
        <div className="bg-white rounded-2xl border border-gray-200 shadow-sm overflow-hidden">
          {/* Purple Header */}
          <div className="bg-gradient-to-r from-indigo-600 to-purple-600 px-6 py-4">
            <h3 className="text-lg font-bold text-white">Attendance Records</h3>
            <p className="text-sm text-indigo-100 mt-0.5">{filteredRecords.length} records found</p>
          </div>

          {/* Table */}
          <div className="overflow-x-auto">
            <table className="w-full">
              <thead>
                <tr className="border-b border-gray-200 bg-gray-50">
                  <th className="px-6 py-3 text-left text-xs font-semibold text-gray-600 uppercase tracking-wider">PARENT</th>
                  <th className="px-6 py-3 text-left text-xs font-semibold text-gray-600 uppercase tracking-wider">CHILDREN</th>
                  <th className="px-6 py-3 text-left text-xs font-semibold text-gray-600 uppercase tracking-wider">ACTION</th>
                  <th className="px-6 py-3 text-left text-xs font-semibold text-gray-600 uppercase tracking-wider">TIME</th>
                  <th className="px-6 py-3 text-left text-xs font-semibold text-gray-600 uppercase tracking-wider">PERFORMED BY</th>
                  <th className="px-6 py-3 text-left text-xs font-semibold text-gray-600 uppercase tracking-wider">STATUS</th>
                  <th className="px-6 py-3 text-left text-xs font-semibold text-gray-600 uppercase tracking-wider">FACIAL CAPTURE</th>
                </tr>
              </thead>
              <tbody className="divide-y divide-gray-200">
                {filteredRecords.map((record: any, index: number) => {
                  const parent = parents.find(p => p.id === record.parentId);
                  const clockInTime = record.clockIn ? new Date(record.clockIn) : null;
                  const clockOutTime = record.clockOut ? new Date(record.clockOut) : null;
                  const isLate = clockInTime && (clockInTime.getHours() > 8 || (clockInTime.getHours() === 8 && clockInTime.getMinutes() > 0));
                  const isIncomplete = record.clockIn && !record.clockOut;

                  // Determine which action to show (prioritize clock-in if both exist)
                  const displayAction = record.clockIn ? 'clock-in' : 'clock-out';
                  const displayTime = displayAction === 'clock-in' ? clockInTime : clockOutTime;
                  const performedByAssignee = displayAction === 'clock-in' ? record.clockInAssigneeId : record.clockOutAssigneeId;
                  const assigneeName = displayAction === 'clock-in' ? record.clockInAssigneeName : record.clockOutAssigneeName;
                  const assigneePhoto = displayAction === 'clock-in' ? record.clockInAssigneePhoto : record.clockOutAssigneePhoto;
                  const faceImage = displayAction === 'clock-in' ? record.clockInFaceImage : record.clockOutFaceImage;

                  return (
                    <tr key={index} className="hover:bg-gray-50 transition-colors">
                      {/* PARENT */}
                      <td className="px-6 py-4">
                        <div className="flex items-center gap-3">
                          <div className="relative group cursor-pointer">
                            <img
                              src={record.parentPhoto || `https://api.dicebear.com/7.x/avataaars/svg?seed=${parent?.name || record.parentName}`}
                              alt={parent?.name || record.parentName}
                              className="w-10 h-10 rounded-full object-cover border-2 border-gray-200"
                              onClick={() => setImageViewer({ 
                                isOpen: true, 
                                imageUrl: record.parentPhoto || `https://api.dicebear.com/7.x/avataaars/svg?seed=${parent?.name || record.parentName}`, 
                                title: parent?.name || record.parentName,
                                subtitle: 'Parent Photo'
                              })}
                              onError={(e) => {
                                e.currentTarget.src = `https://api.dicebear.com/7.x/avataaars/svg?seed=${parent?.name || record.parentName}`;
                              }}
                            />
                          </div>
                          <span className="text-sm font-medium text-gray-900">{parent?.name || record.parentName}</span>
                        </div>
                      </td>

                      {/* CHILDREN */}
                      <td className="px-6 py-4">
                        <div className="flex flex-wrap gap-1">
                          {record.children.map((childName: string, idx: number) => (
                            <span
                              key={idx}
                              className="inline-flex items-center px-2 py-0.5 rounded text-xs font-medium bg-indigo-100 text-indigo-700 cursor-pointer hover:bg-indigo-200 transition-colors"
                            >
                              {childName}
                            </span>
                          ))}
                        </div>
                      </td>

                      {/* ACTION */}
                      <td className="px-6 py-4">
                        <div className="flex items-center gap-2">
                          <div className={`p-1.5 rounded-full ${displayAction === 'clock-in' ? 'bg-blue-100' : 'bg-purple-100'}`}>
                            <Clock className={`w-4 h-4 ${displayAction === 'clock-in' ? 'text-blue-600' : 'text-purple-600'}`} />
                          </div>
                          <span className={`text-sm font-medium ${displayAction === 'clock-in' ? 'text-blue-600' : 'text-purple-600'}`}>
                            {displayAction === 'clock-in' ? 'Clock In' : 'Clock Out'}
                          </span>
                        </div>
                      </td>

                      {/* TIME */}
                      <td className="px-6 py-4">
                        <div>
                          <div className="text-sm font-semibold text-gray-900">
                            {displayTime?.toLocaleTimeString('en-US', { hour: '2-digit', minute: '2-digit' })}
                          </div>
                          <div className="text-xs text-gray-500">
                            {displayTime?.toLocaleDateString('en-US', { month: 'short', day: 'numeric' })}
                          </div>
                        </div>
                      </td>

                      {/* PERFORMED BY */}
                      <td className="px-6 py-4">
                        {performedByAssignee ? (
                          <div className="flex items-center gap-2">
                            <img
                              src={assigneePhoto || `https://api.dicebear.com/7.x/avataaars/svg?seed=${assigneeName}`}
                              alt={assigneeName}
                              className="w-8 h-8 rounded-full object-cover border-2 border-orange-300 cursor-pointer"
                              onClick={() => setImageViewer({ 
                                isOpen: true, 
                                imageUrl: assigneePhoto || `https://api.dicebear.com/7.x/avataaars/svg?seed=${assigneeName}`, 
                                title: assigneeName || 'Assignee',
                                subtitle: 'Assignee Photo'
                              })}
                              onError={(e) => {
                                e.currentTarget.src = `https://api.dicebear.com/7.x/avataaars/svg?seed=${assigneeName}`;
                              }}
                            />
                            <div>
                              <div className="text-sm font-medium text-gray-900">{assigneeName}</div>
                              <div className="text-xs text-gray-500">Assignee</div>
                              <div className="text-xs text-gray-400">by {record.parentName}</div>
                            </div>
                          </div>
                        ) : (
                          <div className="flex items-center gap-2">
                            <img
                              src={record.parentPhoto || `https://api.dicebear.com/7.x/avataaars/svg?seed=${parent?.name || record.parentName}`}
                              alt={parent?.name || record.parentName}
                              className="w-8 h-8 rounded-full object-cover border-2 border-emerald-300"
                            />
                            <div>
                              <div className="text-sm font-medium text-gray-900">{parent?.name || record.parentName}</div>
                              <div className="text-xs text-gray-500">Parent</div>
                            </div>
                          </div>
                        )}
                      </td>

                      {/* STATUS */}
                      <td className="px-6 py-4">
                        <div className="space-y-1">
                          {isLate && (
                            <span className="inline-block px-2 py-1 rounded text-xs font-semibold bg-red-100 text-red-700">
                              Late
                            </span>
                          )}
                          {isIncomplete && (
                            <span className="inline-block px-2 py-1 rounded text-xs font-semibold bg-yellow-100 text-yellow-700">
                              Incomplete
                            </span>
                          )}
                          {!isLate && !isIncomplete && record.clockIn && record.clockOut && (
                            <span className="inline-block px-2 py-1 rounded text-xs font-semibold bg-green-100 text-green-700">
                              Complete
                            </span>
                          )}
                        </div>
                      </td>

                      {/* FACIAL CAPTURE */}
                      <td className="px-6 py-4">
                        {faceImage ? (
                          <button
                            onClick={() => setImageViewer({ 
                              isOpen: true, 
                              imageUrl: faceImage, 
                              title: `${displayAction === 'clock-in' ? 'Clock In' : 'Clock Out'} - Facial Capture`,
                              subtitle: `${performedByAssignee ? assigneeName : record.parentName} • ${displayTime?.toLocaleString()}`
                            })}
                            className="inline-flex items-center gap-1 px-3 py-1.5 bg-indigo-100 hover:bg-indigo-200 text-indigo-700 rounded-lg text-xs font-medium transition-colors"
                          >
                            <Eye className="w-3.5 h-3.5" />
                            View
                          </button>
                        ) : (
                          <span className="text-xs text-gray-400">No capture</span>
                        )}
                      </td>
                    </tr>
                  );
                })}
              </tbody>
            </table>
          </div>

          {/* Footer */}
          <div className="bg-gray-50 px-6 py-3 border-t border-gray-200 flex items-center justify-between">
            <div className="flex items-center gap-2 text-sm text-gray-600">
              <Calendar className="w-4 h-4" />
              <span>{new Date().toLocaleDateString('en-US', { weekday: 'long', month: 'long', day: 'numeric', year: 'numeric' })}</span>
            </div>
            <div className="text-sm font-medium text-gray-700">
              Total Records: <span className="text-indigo-600 font-bold">{filteredRecords.length}</span>
            </div>
          </div>
        </div>
      )}
